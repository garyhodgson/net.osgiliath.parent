44  /**
45   * JMS sample of Hello service exports.
46   * 
47   * @author charliemordant
48   * 
49   */
50  @Slf4j
51  public class HelloServiceJMS implements HelloService, MessageListener {
52    /**
53     * Database persistence repository.
54     */
55    @Setter
56    private transient HelloObjectRepository helloObjectRepository;
57    /**
58     * Hibernate validator.
59     */
60    @Setter
61    private transient Validator validator;
62    /**
63     * JMS producer.
64     */
65    @Setter
66    private transient JmsOperations template;
67  
68    /**
69     * JMS consumer.
70     * 
71     * @throws JMSException persistence error
72     */
73    @Override
74    public void persistHello(HelloEntity hello) {
75  
76      log
77          .error("****************** Save on JMS Service **********************");
78      log
79          .info("persisting new message with jms: " + hello.getHelloMessage());
80      final Set<ConstraintViolation<HelloEntity>> validationResults = this.validator
81          .validate(hello);
82      final StringBuilder errors = new StringBuilder("");
83  
84      if (!validationResults.isEmpty()) {
85        for (final ConstraintViolation<HelloEntity> violation : validationResults) {
86          log.info("subscription error, validating user:"
87              + violation.getMessage());
88          errors.append(violation.getPropertyPath()).append(": ")
89              .append(violation.getMessage().replaceAll("\"", "")).append(';')
90              .append(System.lineSeparator());
91        }
92        this.template.send("helloServiceQueueErrors", new MessageCreator() {
93          /**
94           * Creates a JMS message
95           * @param session JMS session
96           * @return the JMS message
97           * @throws JMSException invalid payload
98           */
99          public Message createMessage(final Session session) throws JMSException {          
100           return session.createTextMessage(errors.toString());
101         }
102       });
103     }
104     this.helloObjectRepository.save(hello);
105     this.template.send("helloServiceQueueOut", new MessageCreator() {
106       /**
107        * Creates a JMS message
108        * @param session JMS session
109        * @return the JMS message
110        * @throws JMSException invalid payload
111        */
112       public Message createMessage(final Session session) throws JMSException {
113         return session.createObjectMessage(getHellos());
114       }
115     });
116 
117   }
118 
119   /**
120    * Returns all entities.
121    * @return all entities
122    */
123   @Override
124   public Hellos getHellos() {
125 
126     final Collection<HelloEntity> helloObjects = this.helloObjectRepository
127         .findAll();
128     if (helloObjects.isEmpty()) {
129       throw new UnsupportedOperationException(
130           "You could not call this method when the list is empty");
131     }
132     return Hellos
133         .builder()
134         .helloCollection(
135             Lists.newArrayList(Iterables.transform(helloObjects,
136                 this.helloObjectToStringFunction))).build();
137   }
138 
139   /**
140    * Function that transforms entities to string.
141    */
142   private final transient Function<HelloEntity, String> helloObjectToStringFunction = new Function<HelloEntity, String>() {
143 
144     @Override
145     public String apply(HelloEntity entity) {
146       return entity.getHelloMessage();
147     }
148   };
149 
150   /**
151    * Deletes all entities.
152    */
153   @Override
154   public void deleteAll() {
155     this.helloObjectRepository.deleteAll();
156 
157   }
158   /**
159    * Method called on message reception.
160    * @param message received message
161    */
162   @Override
163   public void onMessage(Message message) {
164     try {
165       final HelloEntity helloObject_p = (HelloEntity) ((ObjectMessage) message)
166           .getObject();
167       this.persistHello(helloObject_p);
168     }
169     catch (JMSException e) {
170       log.error("error receiving JMS message", e);
171     }
172 
173   }
174 }
